Getting Started

 

Description :

 

 This Application provides comprehensive details about diverse human diseases classified in categories & subcategories, diseases range from physical to mental diseases with diverse types of diseases available for the user.
Features:
- Comprehensive information about diseases classified in Categories & Sub-Categories
- Search & Favorite list
- News
- Related articles & keywords links

Home page code description

$(document).on("pagebeforeshow", "#CurrentPage", function (e) { ini();//increasing performance LoadNews();//start loading news }); $(document).bind("mobileinit", function () { $.mobile.ajaxLinksEnabled = false; ini(); }); //increasing performance function ini() { $.mobile.defaultPageTransition = 'none'; $.mobile.defaultDialogTransition = 'none'; $.mobile.useFastClick = true; $.mobile.buttonMarkup.hoverDelay = 0; } var news; function LoadNews() { //ajax call to connect news headlines feeds service $(function () { url = 'http://www.medpagetoday.com/rss/Headlines.xml'; $.ajax({ type: "GET", url: 'http://ajax.googleapis.com/ajax/services/feed/load?v=1.0&num=1000&callback=?&q=' + encodeURIComponent(url), dataType: 'json', async: true, error: function () { }, success: function (xml) { values = xml.responseData.feed.entries;//load response data into values array news = values; StartNewsLoading();//start showing news } }); }); } function StartNewsLoading() { try { var i = -1; //timer working every 2 s to change current news title setInterval(function () { if (i == -1) { $("#currentNews").html("<b><h2>Latest News</h2></b>"); $("#currentNews").height(40); } else { $("#currentNews").html(news[i]['title']);//set current news title } i++; if (i == news.length) { i = -1; } }, 2000); } catch (err) { } }

Categories page code description

$(document).on("pagebeforeshow", "#CurrentPage", function (e) { ini(); LoadCat();//loading categories }); $(document).bind("mobileinit", function () { $.mobile.ajaxLinksEnabled = false; ini(); }); //increasing performance function ini() { $.mobile.defaultPageTransition = 'none'; $.mobile.defaultDialogTransition = 'none'; $.mobile.useFastClick = true; $.mobile.buttonMarkup.hoverDelay = 0; } function LoadCat() { try { var id = getParameterByName("id");//get category id parameter from url var url = "data/cat/" + id + ".txt"; //ajax call to get categories titles $.ajax({ url: url, dataType: "text", type: "GET", async: true, success: function (data) { var cats = data.split('\n'); var oput = ""; for (var c = 0; c < cats.length; c++) { var cat = cats[c]; var li = "<li><a data-transition='none' data-ajax='false' href='topics.html?id=" + cat.split('@')[1] + "'>" + "<h2>" + cat.split('@')[0] + "</h2>" + "</a></li>"; oput += li; } $("#lstCats").append(oput);//adding items to the categories listview $("#lstCats").listview("refresh");//rendering listview } }); } catch (err) { } } //get parameters from url function getParameterByName(name) { name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]"); var regexS = "[\\?&]" + name + "=([^&#]*)"; var regex = new RegExp(regexS); var results = regex.exec(window.location.search); if (results == null) return ""; else return decodeURIComponent(results[1].replace(/\+/g, " ")); }
